###### CREATE FEATURING  (POST) ######



curl --location 'http://localhost:3000/api/v1/featurings' \
--header 'Username: admin' \
--header 'Password: 123' \
--header 'Content-Type: application/json' \
--data '{
    "featuring_name": "TESTE",
    "is_album": true,
    "artist": "TESTE",
    "label": "Pug Records",
    "info": null,
    "release_date": "2018-11-07",
    "credit": [
        "Producer"
    ]
}'

##### DELETE FEATURING (DELETE)  #########
